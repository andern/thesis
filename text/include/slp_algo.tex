\begin{algorithm}[ht!]
\caption{An Iterative Method}
\label{alg:iter}
\DontPrintSemicolon
%\LinesNumbered
Given $x_0$ and some $\epsilon \ge 0$\;
Set $k \leftarrow 0$\;

\Repeat{$\frac{f(x_{k-1}) - f(x_k)}{|f(x_{k-1})|} \leq \epsilon$} {
    $T_k \leftarrow - x_k^THx_k + 2x_k^THx + b^Tx$\;
    Solve $\mathcal{L}_k$ and call its optimal solution $\hat{x}_k$\;
    $\alpha_k = \displaystyle
                \frac{
                      2\hat{x}_k^T H \hat{x}_k
                    - 2x_k^T H \hat{x}_k
                    + b^T \hat{x}_k
                    - b^T x_k
                     }{
                      2x_k^THx_k
                    - 4x_k^TH\hat{x}_k
                    + 2\hat{x}_k^T H \hat{x}_k
                     }$\;
%   $\alpha_k \in \argmin_{\alpha \leq 1} f(\alpha x_k + (1-\alpha)\hat{x}_k)$\;
    $x_{k+1} = \alpha_k x_k + (1-\alpha_k)\hat{x}_k$\;
    $k \leftarrow k +1$\;
}
\end{algorithm}
