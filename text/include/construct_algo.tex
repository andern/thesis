\begin{algorithm}[H]
\caption{$construct(\mathcal{Q}, b)$}
\label{alg:construct}
$V \leftarrow \{v_0\}$ \\
$E \leftarrow \emptyset$ \\
\SetAlgoVlined

\ForEach{modifier $\mathcal{M}_i \in \mathcal{P}(\left\{{1,2,\ldots,n}\right\})
, |\mathcal{M}_i| \leq b$}{
    $(a, k) \leftarrow find(\mathcal{M}_i, v_0) = -1$\\
    \If{a \textrm{is false}}{
        $V \leftarrow V \cup \{v_i\}$ \\
        $E \leftarrow E \cup \{(k, i)\}$ \\
    }
}
$G \leftarrow (V, E)$ \\
\KwRet{$G$}
\end{algorithm}
